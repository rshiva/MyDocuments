DELIMITER //
	CREATE PROCEDURE test_data
	(rows INT)
	BEGIN
		DECLARE val1 FLOAT;
		DECLARE val2 FLOAT;
		REPEAT
		SELECT RAND() INTO val1;
		SELECT RAND() INTO val2;
		INSERT INTO analysis VALUES(NULL, val1, val2);
		SET rows = rows - 1;
		UNTIL rows = 0
		END REPEAT;
	END//
DELIMITER ;
